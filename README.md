# Dependencies
    npx @angular/cli add @angular/material
    npx @angular/cli add @angular/fire
    npm i -s @angular/flex-layout
    npm install --save hammerjs
    npm install moment
    npm i uuid
    npm i @types/uuid
    npm install ngx-toastr
    ag-grid-community ag-grid-angular
